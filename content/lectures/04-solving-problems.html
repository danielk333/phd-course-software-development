                        def dot(vec1, vec2):
                            assert len(vec1) == len(vec2), "Input vectors must be same length"
                            if len(vec1) == 0:
                                return
                            common_iter = zip(vec1, vec2)

                            x1, x2 = next(common_iter)
                            result = x1 * x2

                            for x1, x2 in common_iter:
                                result += x1 * x2
                            
                            return result

                        from numpy.typing import NDArray

                        def dot(vec1: NDArray, vec2: NDArray):
                            assert vec1.dtype == vec2.dtype, "Input vectors must be same type"
                            assert vec1.size == vec2.size, "Input vectors must be same size"
                            if vec1.size == 0:
                                return

                            result = vec1.dtype.type(0)
                            for x1, x2 in zip(vec1, vec2):
                                result += x1 * x2
                            
                            return result
